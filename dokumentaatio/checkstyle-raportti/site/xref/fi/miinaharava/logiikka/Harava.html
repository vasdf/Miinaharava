1   package fi.miinaharava.logiikka;

3   public class Harava {
4       private Ruutu[][] kentta;
5       
6       public Harava(Ruutu[][] kentta) {
7           this.kentta = kentta;
8       }
9   
10      public void haravoiRuutu(Ruutu ruutu) {
11          ruutu.haravoi();
12  
13          haravoiRuudutYmparilta(ruutu);
14      }
15  
16      public void haravoiRuudutYmparilta(Ruutu ruutu) {
17          for (int y = 0; y < kentta.length; y++) {
18              for (int x = 0; x < kentta.length; x++) {
19                  if (onkoLahekkain(ruutu, kentta[y][x]) && ruutu != kentta[y][x]) {
20                      if (ruutu.getArvo() == 0 && kentta[y][x].onkoHaravoitu() == false) {
21                          haravoiRuutu(kentta[y][x]);
22                      } else if (kentta[y][x].getArvo() == 0 && kentta[y][x].onkoHaravoitu() == false) {
23                          haravoiRuutu(kentta[y][x]);
24                      }
25                  }
26              }
27          }
28      }
29  
30      public boolean onkoLahekkain(Ruutu ruutu1, Ruutu ruutu2) {
31          return Math.abs(ruutu1.getX() - ruutu2.getX()) <= 1 && Math.abs(ruutu1.getY() - ruutu2.getY()) <= 1;
32      }
33  }
