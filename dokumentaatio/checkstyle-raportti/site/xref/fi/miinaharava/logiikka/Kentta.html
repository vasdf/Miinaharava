1   package fi.miinaharava.logiikka;
2   
3   public class Kentta {
4       
5       private Ruutu[][] ruudukko;
6       private int sivunpituus;
7       private int miinojenmaara;
8       private Miinojenluoja miinojenluoja;
9       private Harava harava;
10  
11      public Kentta(int sivunpituus, int miinojenmaara) {
12          this.sivunpituus = sivunpituus;
13          this.miinojenmaara = miinojenmaara;
14          this.ruudukko = new Ruutu[sivunpituus][sivunpituus];
15          this.harava = new Harava(ruudukko);
16          
17          for (int y = 0; y < sivunpituus; y++) {
18              for (int x = 0; x < sivunpituus; x++) {
19                  Ruutu ruutu = new Ruutu(x, y);
20                  ruudukko[y][x] = ruutu;
21              }
22          }
23          
24          miinojenluoja = new Miinojenluoja(sivunpituus, miinojenmaara, ruudukko);
25          ruudukko = miinojenluoja.luoMiinatSatunnaisesti();
26      }
27      
28      public void haravoi(Ruutu ruutu) {
29          harava.haravoiRuutu(ruutu);;
30      }
31      
32      public Ruutu[][] getKentta() {
33          return ruudukko;
34      }
35      
36      public Miinojenluoja getMiinojenluoja() {
37          return miinojenluoja;
38      }
39  
40  }
