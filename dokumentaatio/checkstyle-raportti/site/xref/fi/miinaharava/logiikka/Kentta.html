1   package fi.miinaharava.logiikka;
2   
3   import java.awt.Font;
4   import java.awt.Insets;
5   import javax.swing.JButton;
6   
7   /**
8    * Luokka luo Ruutu olioista koostuvan taulun eli pelikentän.
9    */
10  public class Kentta {
11  
12      private Ruutu[][] ruudukko;
13      private int sivunpituus;
14      private int miinojenmaara;
15      private Miinojenluoja miinojenluoja;
16      private RuudunArvonlaskija ruudunarvonlaskija;
17      private RuutujenPainaja painaja;
18  
19      /**
20       * Konstruktorissa luodaan halutun lainen Ruuduista koostuva taulukko, joka
21       * toimii kenttänä.
22       *
23       * @param sivunpituus Käyttäjän haluama sivunpituus
24       *
25       * @param miinojenmaara Käyttäjän haluama miinojenmäärä
26       */
27      public Kentta(int sivunpituus, int miinojenmaara) {
28          this.sivunpituus = sivunpituus;
29          this.miinojenmaara = miinojenmaara;
30          this.ruudukko = new Ruutu[sivunpituus][sivunpituus];
31  
32          luoPelialusta();
33      }
34      
35      /**
36       * Metodi luo Ruutu olioille tehtyyn taulukkoon Ruutu olioit.
37       */
38      private void luoPelialusta() {
39          for (int y = 0; y < sivunpituus; y++) {
40              for (int x = 0; x < sivunpituus; x++) {
41                  Ruutu ruutu = new Ruutu(x, y);
42  
43                  JButton painike = new JButton();
44                  
45                  ruutu.setPainike(painike);
46  
47                  ruudukko[y][x] = ruutu;
48              }
49          }
50          
51          miinojenluoja = new Miinojenluoja(sivunpituus, miinojenmaara);
52          ruudukko = miinojenluoja.luoMiinatSatunnaisesti(ruudukko);
53  
54          ruudunarvonlaskija = new RuudunArvonlaskija();
55          ruudukko = ruudunarvonlaskija.laskeArvot(ruudukko);
56  
57          painaja = new RuutujenPainaja(ruudukko);
58      }
59  
60      /**
61       * Metodi kutsuu VierekkaisteRuutujenPainajan metodia paina.
62       *
63       * @param ruutu TapahtumaKuuntelijalta saatu Ruutu, jota painettiin
64       */
65      public void paina(Ruutu ruutu) {
66          painaja.paina(ruutu);
67      }
68  
69      public Ruutu[][] getKentta() {
70          return ruudukko;
71      }
72      
73      /**
74       * Metodi on testejä varten.
75       * 
76       * @return palauttaa miinojeluojaolion
77       */
78      public Miinojenluoja getMiinojenluoja() {
79          return miinojenluoja;
80      }
81  }
