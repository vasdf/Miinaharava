1   package fi.miinaharava.logiikka;
2   
3   import java.util.Random;
4   
5   /**
6    * Luokka luo valmiiksi luotuun Ruutu olioita sisältävään taulukkoon
7    * miinat eli muuttaa Ruutu olion boolean arvon miina todeksi.
8    */
9   public class Miinojenluoja {
10  
11      private int sivunpituus;
12      private int miinojenmaara;
13      private Random random;
14      private Ruutu[][] ruudukko;
15      /**
16       * Konstruktorissa kerrotaan miinojenluojalle tarvitut arvot.
17       * 
18       * @param sivunpituus   Kentän antama sivunpituus
19       * 
20       * @param miinojenmaara Kentän antama miinojenmäärä
21       */
22      public Miinojenluoja(int sivunpituus, int miinojenmaara) {
23          this.sivunpituus = sivunpituus;
24          this.miinojenmaara = miinojenmaara;;
25          this.random = new Random();
26      }
27      /**
28       * Metodi muuttaa parametrinä saamansa
29       * Ruutu olioita sisältävän taulun Ruutuja miinoiksi
30       * eli asettaa Ruutu olion boolean arvon miina todeksi
31       * 
32       * Ruutuja muutetaan konstruktorissa annettu miinojemaara.
33       * 
34       * @param ruudukko  Kentta luokan luoma ruudukko
35       * 
36       * @return taulukko, jossa Ruutuja on muunnettu miinoiksi haluttu määrä
37       */
38      public Ruutu[][] luoMiinatSatunnaisesti(Ruutu[][] ruudukko) {
39          this.ruudukko = ruudukko;
40          
41          int miinojaluotu = 0;
42  
43          while (miinojaluotu < miinojenmaara) {
44              int x = random.nextInt(sivunpituus);
45              int y = random.nextInt(sivunpituus);
46  
47              if (ruudukko[y][x].onkoMiina() == false) {
48                  ruudukko[y][x].asetaMiina();
49                  
50                  miinojaluotu++;
51              }
52          }
53  
54          return ruudukko;
55      }
56  }
